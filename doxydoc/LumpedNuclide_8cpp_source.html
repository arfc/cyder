454 mat_rsrc_ptr LumpedNuclide::extractIntegratedMass(NuclideModelPtr daughter,
455  double the_time){

457  double del_r = geom()->outer_radius() - geom()->inner_radius();
458  double len = geom()->length();
459  // scalar = 2*pi*l*theta*(r_j-r_i)^2

465  IsoConcMap c_i_scaled = MatTools::scaleConcMap(c_i_n, 0.5*scalar);
466  IsoConcMap scaled_concs = MatTools::addConcMaps(c_j_scaled, c_i_scaled);
