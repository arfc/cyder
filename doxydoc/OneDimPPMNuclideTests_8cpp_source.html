64 double OneDimPPMNuclideTest::calculate_conc(IsoConcMap C_0, IsoConcMap C_i, double r, Iso iso, int t0, int t){
65  double D = mat_table_->D(iso/1000);
66  double pi = boost::math::constants::pi<double>();
67  //@TODO add sorption to this model. For now, R=1, no sorption.
68  double R=1;
69  //@TODO convert months to seconds
70 
71  double At_frac = (R*r - v_*t)/(2*pow(D*R*t, 0.5));
72  double At = 0.5*boost::math::erfc(At_frac);
73  double Att0_frac = (R*r - v_*t)/(2*pow(D*R*(t-t0), 0.5));
74  double Att0 = 0.5*boost::math::erfc(Att0_frac);
75  double A = At - Att0;
76  double B1_frac = (R*r - v_*t)/(2*pow(D*R*t, 0.5));
77  double B1 = 0.5*boost::math::erfc(B1_frac);
78  double B2_exp = -pow(R*r-v_*t,2)/(4*D*R*t);
79  double B2 = pow(v_*v_*t/(pi*R*D),0.5)*exp(B2_exp);
80  double B3_exp = v_*r/D;
81  double B3_frac = (R*r + v_*t)/(2*pow(D*R*t,0.5));
82  double B3_factor = -0.5*(1+(v_*r)/(D) + (v_*v_*t)/(D*R));
83  double B3 = B3_factor*exp(B3_exp)*boost::math::erfc(B3_frac); ;
84  double B = C_i[iso]*(B1 + B2 + B3);
85 
86  double to_ret = C_0[iso]*A + B;
87 
88  return to_ret;
89 }
